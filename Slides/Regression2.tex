\section{GLM}
\frame{\tableofcontents[currentsection]}

\frame{
\frametitle{Binary Outcome Models}
\begin{itemize}\itemsep2em
\item GLMs are estimated with \texttt{glm}
\item Works basically the same as \texttt{lm}
\item Additional argument: \texttt{family}
	\begin{itemize}
	\item gaussian
	\item binomial
	\item poisson
	\item Various others
	\end{itemize}
\end{itemize}
}

\frame{
\frametitle{Note on other GLMs}
\begin{itemize}\itemsep2em
\item Ordered models: \textbf{library(MASS)}
\item Multinomial logit: \textbf{library(nnet)} and \textbf{library(mlogit)}
\item The \textbf{Zelig} package provides additional models
\end{itemize}
}
